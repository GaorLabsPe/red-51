        Paso 1: Setup de Base de Datos (Solo una vez)

        Para una nueva tienda, primero debes crear las tablas en tu base de datos. Ve a tu proyecto de Supabase > SQL Editor > "New query", pega el siguiente script completo y haz clic en "RUN".

        Ver/Ocultar Script SQL Completo 
        -- 1. CREACIN DE TABLAS
CREATE TABLE public.categorias ( 
  id bigint generated by default as identity primary key, 
  nombre text not null, 
  slug text not null unique, 
  icon text, 
  imagen_url text 
);

CREATE TABLE public.productos ( 
  id bigint generated by default as identity primary key, 
  nombre text not null, 
  marca text not null, 
  categoria_id bigint references public.categorias(id) on delete set null, 
  descripcion text, 
  precio numeric not null, 
  stock integer not null default 0, 
  icon text, 
  badge text, 
  activo boolean default true, 
  imagen_url text, 
  variantes jsonb 
);

CREATE TABLE public.pedidos ( 
  id uuid default gen_random_uuid() primary key, 
  fecha_pedido timestamp with time zone default timezone('utc'::text, now()) not null, 
  numero_pedido text, 
  nombre_cliente text not null, 
  telefono_cliente text not null, 
  email_cliente text, 
  direccion text, 
  notas text, 
  latitud double precision, 
  longitud double precision, 
  productos jsonb, 
  total numeric not null, 
  cantidad_items integer, 
  estado text default 'pendiente_pago'::text not null 
);

CREATE TABLE public.settings (
  id bigint PRIMARY KEY,
  config jsonb
);

-- 2. HABILITAR ROW LEVEL SECURITY (RLS) - MUY IMPORTANTE PARA LA SEGURIDAD!
ALTER TABLE public.categorias ENABLE ROW LEVEL SECURITY; 
ALTER TABLE public.productos ENABLE ROW LEVEL SECURITY; 
ALTER TABLE public.pedidos ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.settings ENABLE ROW LEVEL SECURITY;

-- 3. POLTICAS DE ACCESO PBLICO (PERMITEN QUE CUALQUIER VISITANTE VEA LA TIENDA Y HAGA PEDIDOS)
CREATE POLICY "Public categories are viewable" ON public.categorias FOR SELECT USING (true);
CREATE POLICY "Public active products are viewable" ON public.productos FOR SELECT USING (activo = true);
CREATE POLICY "Allow public insert for orders" ON public.pedidos FOR INSERT WITH CHECK (true);
CREATE POLICY "Public settings are viewable" ON public.settings FOR SELECT USING (true);

-- 4. POLTICAS DE ACCESO PARA ADMIN (ACCESO TOTAL PARA USUARIOS AUTENTICADOS)
CREATE POLICY "Allow full access for authenticated users" ON public.categorias FOR ALL USING (auth.role() = 'authenticated');
CREATE POLICY "Allow full access for authenticated users" ON public.productos FOR ALL USING (auth.role() = 'authenticated');
CREATE POLICY "Allow full access for authenticated users" ON public.pedidos FOR ALL USING (auth.role() = 'authenticated');
CREATE POLICY "Allow full access for authenticated users" ON public.settings FOR ALL USING (auth.role() = 'authenticated');

        Paso 2: Creacin de Usuarios (Mtodo Seguro)

        Importante: Por seguridad, los usuarios deben crearse desde el panel de Supabase para que las contraseas se almacenen de forma encriptada. No uses scripts SQL para esto.

        Ve a tu panel de Supabase > Authentication > Users y crea las siguientes cuentas:

        • Usuario Root: root@tienda.com (Para tu acceso de super-administrador).
        • Dueo del Negocio: admin@negocio.com
        • Personal de Ventas 1: ventas1@negocio.com
        • Personal de Ventas 2: ventas2@negocio.com

        Nota: Actualmente, todas las cuentas creadas tienen los mismos permisos de administrador.
